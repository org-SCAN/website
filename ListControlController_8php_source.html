182 public function storeFields(StoreListControlFieldsRequest $request, ListControl $listControl)
183 {
184 $this->authorize('create', $listControl);
185 //store the fields onto the ListStructure table
186 foreach($request->validated()['fields'] as $field){
187 if(!empty($field) && ($listControl->structure->first() == null || !$listControl->structure()->where('field', $field['name'])->exists())){
188 $listControl->structure()->create([
189 "field" => Str::snake($field['name']),
190 "data_type_id" => $field['data_type_id'],
191 "required" => $field['required'] ?? 0,
192 ]);
193 }
194 }
195
196 Artisan::call("make:list", ["id"=>$listControl->id]);
197 Artisan::call("migrate");
198
199 $listControl->refresh();
200 $fields = $listControl->structure;
201 return view("lists_control.define_displayed_value", compact("listControl", "fields"));
